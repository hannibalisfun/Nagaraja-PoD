# FEATURE 1: PURGE THE SERPENT FROM INDIA
# Events podnagaraja.1-2
#####################################################

# Victory Celebration
# Triggered by: purge_serpent_india_decision effect block
podnagaraja.1 = {
	type = character_event
	window = fullscreen_event
	title = podnagaraja.1.t
	desc = podnagaraja.1.desc
	theme = vampire
	override_background = { reference = throne_room }

	immediate = {
	}

	option = {
		name = podnagaraja.1.a

		add_character_modifier = {
			modifier = scourge_of_sutekh_modifier
		}
		add_prestige = 1500
		root.dynasty = {
			add_dynasty_prestige = 1000
		}

		# Setite/Daitya rulers hate you
		hidden_effect = {
			every_ruler = {
				limit = {
					has_trait = setite
				}
				add_opinion = {
					modifier = hate_opinion
					target = scope:serpent_purger
					opinion = -100
				}
			}
		}

		# Notify other players in India/Tibet
		hidden_effect = {
			every_player = {
				limit = {
					NOT = { this = root }
					capital_province = {
						OR = {
							geographical_region = world_india
							geographical_region = world_tibet
						}
					}
				}
				trigger_event = podnagaraja.2
			}
		}
	}
}

# FEATURE 2: PILGRIMAGE TO ENOCH (Journey)
# Events podnagaraja.10-21
# Started via pilgrimage_to_enoch_decision -> Journey system
# Travel phase triggers podnagaraja.10 on arrival
#
# Flow:
#   10 (arrival/setup) -> 11 (dispatch) -> [pool 12-19] -> 11 -> ... -> 20 (final)
#   Pool: 8 events, 4-5 randomly selected per journey
#   21: Fellow Traveler spy follow-up
#####################################################

# ===== ARRIVAL & DISPATCH =====

# Arrival event — sets up the journey and chains to dispatch
# Triggered by: pilgrimage_to_enoch_travel_phase SGUI
podnagaraja.10 = {
	type = character_event
	title = podnagaraja.10.t
	desc = podnagaraja.10.desc
	theme = vampire
	override_background = { reference = wilderness }

	left_portrait = {
		character = root
		animation = personality_content
	}

	immediate = {
		POD_pause_journey_travel_plan_effect = yes
		hidden_effect = {
			# Initialize event counter
			set_variable = {
				name = pilgrimage_events_fired
				value = 0
			}
			# Randomly decide how many pool events (4 or 5)
			random_list = {
				50 = {
					set_variable = {
						name = pilgrimage_events_target
						value = 4
					}
				}
				50 = {
					set_variable = {
						name = pilgrimage_events_target
						value = 5
					}
				}
			}
		}
	}

	option = {
		name = podnagaraja.10.a
		custom_tooltip = podnagaraja.10.a.tt
		hidden_effect = {
			trigger_event = {
				id = podnagaraja.12
				days = { 30 60 }
			}
		}
	}
}

# Hidden dispatch event — picks next random pool event or fires final
podnagaraja.11 = {
	hidden = yes

	trigger = {
		is_alive = yes
	}

	immediate = {
		if = {
			limit = {
				var:pilgrimage_events_fired >= var:pilgrimage_events_target
			}
			# All pool events done — fire the final Gates of Enoch event
			trigger_event = {
				id = podnagaraja.20
				days = { 60 120 }
			}
		}
		else = {
			# Pick a random unfired pool event
			random_list = {
				# Entering the Shadowlands

				# The Tempest
				10 = {
					trigger = { NOT = { has_character_flag = pilgrimage_evt_tempest } }
					add_character_flag = pilgrimage_evt_tempest
					trigger_event = {
						id = podnagaraja.13
						days = { 30 90 }
					}
				}
				# Whispers of the Drowned
				10 = {
					trigger = { NOT = { has_character_flag = pilgrimage_evt_whispers } }
					add_character_flag = pilgrimage_evt_whispers
					trigger_event = {
						id = podnagaraja.14
						days = { 30 90 }
					}
				}
				# The False Necropolis (blocked if warned by wraiths)
				10 = {
					trigger = {
						NOT = { has_character_flag = pilgrimage_evt_false_necropolis }
						NOT = { has_character_flag = nagaraja_wraith_market_warned }
					}
					add_character_flag = pilgrimage_evt_false_necropolis
					trigger_event = {
						id = podnagaraja.15
						days = { 30 90 }
					}
				}
				# A Spectral Anomaly
				10 = {
					trigger = { NOT = { has_character_flag = pilgrimage_evt_anomaly } }
					add_character_flag = pilgrimage_evt_anomaly
					trigger_event = {
						id = podnagaraja.16
						days = { 30 90 }

				}
				# The Wraith Market
				10 = {
					trigger = { NOT = { has_character_flag = pilgrimage_evt_wraith_market } }

						id = podnagaraja.17
