
# Scripted GUIs for the Pilgrimage to Enoch journey
# Each journey needs 3 scripted GUIs:
# [name]_requirements: is_shown for visibility, is_valid for whether the Journey can be started
# [name]_on_end: called when planning reaches 100%, sets destination
# [name]_travel_phase: called when starting travel phase, triggers arrival event

pilgrimage_to_enoch_requirements = {
	scope = character
	is_shown = {
		is_nagaraja_trigger = yes
		religion = religion:talmahera_religion
		NOT = { has_character_flag = pilgrimage_to_enoch_cooldown }
	}
	is_valid = {
		dynasty = {
			has_dynasty_perk = nagaraja_legacy_3
		}
		has_trait = necromancyadvanced
		is_imprisoned = no
		is_at_war = no
	}
}

pilgrimage_to_enoch_on_end = {
	scope = character
	effect = {
		# Spiritual journey to the Underworld - destination is capital
		POD_journey_select_capital_destination_effect = yes
	}
}

pilgrimage_to_enoch_travel_phase = {
	scope = character
	effect = {
		POD_journey_start_travel_phase = {
			EVENT = podnagaraja.10
		}
	}
}
